    Example:

    
class Foo {...}

class Bar {
  @inject()
  private foo

  constructor() {}
}

class TestModule extends AbstractModule {
  public configure() {
    this.bind('foo').to(Foo);
  }
}

var injector = new Injector([new TestModule()]);

//This call inject foo property of class Bar to module defined value.
injector.inject<Bar>(Bar);
